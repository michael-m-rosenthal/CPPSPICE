75 /*
76  Parameter declarations
77  ======================
78 
79  DSK type 2 plate expansion factor
80  ---------------------------------
81 
82  The factor SPICE_DSK_XFRACT is used to slightly expand plates
83  read from DSK type 2 segments in order to perform ray-plate
84  intercept computations.
85 
86  This expansion is performed to prevent rays from passing through
87  a target object without any intersection being detected. Such
88  "false miss" conditions can occur due to round-off errors.
89 
90  Plate expansion is done by computing the difference vectors
91  between a plate's vertices and the plate's centroid, scaling
92  those differences by (1 + SPICE_DSK_XFRACT), then producing new
93  vertices by adding the scaled differences to the centroid. This
94  process doesn't affect the stored DSK data.
95 
96  Plate expansion is also performed when surface points are mapped
97  to plates on which they lie, as is done for illumination angle
98  computations.
99 
100  This parameter is user-adjustable.
101  */
102  #define SPICE_DSK_XFRACT ( 1.0e-10 )
103 
104  /*
105  The keyword for setting or retrieving this factor is
106  */
107  #define SPICE_DSK_KEYXFR 1
108 
109 
110  /*
111  Greedy segment selection factor
112  -------------------------------
113 
114  The factor SGREED is used to slightly expand DSK segment
115  boundaries in order to select segments to consider for
116  ray-surface intercept computations. The effect of this factor is
117  to make the multi-segment intercept algorithm consider all
118  segments that are sufficiently close to the ray of interest, even
119  if the ray misses those segments.
120 
121  This expansion is performed to prevent rays from passing through
122  a target object without any intersection being detected. Such
123  "false miss" conditions can occur due to round-off errors.
124 
125  The exact way this parameter is used is dependent on the
126  coordinate system of the segment to which it applies, and the DSK
127  software implementation. This parameter may be changed in a
128  future version of SPICE.
129  */
130  #define SPICE_DSK_SGREED ( 1.0e-8 )
131 
132  /*
133  The keyword for setting or retrieving this factor is
134  */
135  #define SPICE_DSK_KEYSGR ( SPICE_DSK_KEYXFR + 1 )
136 
137 
138  /*
139  Segment pad margin
140  ------------------
141 
142  The segment pad margin is a scale factor used to determine when a
143  point resulting from a ray-surface intercept computation, if
144  outside the segment's boundaries, is close enough to the segment
145  to be considered a valid result.
146 
147  This margin is required in order to make DSK segment padding
148  (surface data extending slightly beyond the segment's coordinate
149  boundaries) usable: if a ray intersects the pad surface outside
150  the segment boundaries; the pad is useless if the intercept is
151  automatically rejected.
152 
153  However, an excessively large value for this parameter is
154  detrimental, since a ray-surface intercept solution found "in" a
155  segment will supersede solutions in segments farther from the ray's
156  vertex. Solutions found outside of a segment thus can mask solutions
157  that are closer to the ray's vertex by as much as the value of this
158  margin, when applied to a segment's boundary dimensions.
159  */
160  #define SPICE_DSK_SGPADM ( 1.0e-10 )
161 
162 
163  /*
164  The keyword for setting or retrieving this factor is
165  */
166  #define SPICE_DSK_KEYSPM ( SPICE_DSK_KEYSGR + 1 )
167 
168 
169  /*
170  Surface-point membership margin
171  -------------------------------
172 
173  The surface-point membership margin limits the distance
174  between a point and a surface to which the point is
175  considered to belong. The margin is a scale factor applied
176  to the size of the segment containing the surface.
177 
178  This margin is used to map surface points to outward
179  normal vectors at those points.
180 
181  If this margin is set to an excessively small value,
182  routines that make use of the surface-point mapping won't
183  work properly.
184  */
185  #define SPICE_DSK_PTMEMM ( 1.0e-7 )
186 
187  /*
188  The keyword for setting or retrieving this factor is
189  */
190  #define SPICE_DSK_KEYPTM ( SPICE_DSK_KEYSPM + 1 )
191 
192 
193  /*
194  Angular rounding margin
195  -----------------------
196 
197  This margin specifies an amount by which angular values
198  may deviate from their proper ranges without a SPICE error
199  condition being signaled.
200 
201  For example, if an input latitude exceeds pi/2 radians by a
202  positive amount less than this margin, the value is treated as
203  though it were pi/2 radians.
204 
205  Units are radians.
206  */
207  #define SPICE_DSK_ANGMRG ( 1.0e-12 )
208 
209  /*
210  This parameter is not user-adjustable.
211 
212  The keyword for retrieving this parameter is
213  */
214  #define SPICE_DSK_KEYAMG ( SPICE_DSK_KEYPTM + 1 )
215 
216 
217  /*
218  Longitude alias margin
219  ----------------------
220 
221  This margin specifies an amount by which a longitude
222  value can be outside a given longitude range without
223  being considered eligible for transformation by
224  addition or subtraction of 2*pi radians.
225 
226  A longitude value, when compared to the endpoints of
227  a longitude interval, will be considered to be equal
228  to an endpoint if the value is outside the interval
229  differs from that endpoint by a magnitude less than
230  the alias margin.
231 
232 
233  Units are radians.
234  */
235  #define SPICE_DSK_LONALI ( 1.0e-12 )
236 
237  /*
238  This parameter is not user-adjustable.
239 
240  The keyword for retrieving this parameter is
241  */
242  #define SPICE_DSK_KEYLAL ( SPICE_DSK_KEYAMG + 1 )
